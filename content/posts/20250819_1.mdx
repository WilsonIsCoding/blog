## 題解

```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */
var numberOfArithmeticSlices = function (nums) {
    let count = 0;
    let result = 0;
    let currentDiffer = nums[1] - nums[0];
    for (let i = 2; i < nums.length; i++) {
        if (nums[i] - nums[i - 1] === currentDiffer) {
            count++
            result += count
        } else {
            count = 0;
            currentDiffer = nums[i] - nums[i - 1]
        }
    }
    return result;
};
```

時間複雜度：O(n)
空間複雜度：O(3)
