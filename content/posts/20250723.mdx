## 題解

```javascript
/**
 * @param {string} s
 * @param {number} x
 * @param {number} y
 * @return {number}
 */
var maximumGain = function (s, x, y) {
    let fitstCompose = x > y ? 'ab' : 'ba';
    let higherScore = x > y ? x : y;
    let secondCompose = x > y ? 'ba' : 'ab';
    let lowerScore = x > y ? y : x;
    let stack = [];
    let score = 0;
    for (let i = 0; i < s.length; i++) {
        if (stack[stack.length - 1] === fitstCompose[0] && s[i] === fitstCompose[1]) {
            stack.pop()
            score += higherScore
        } else {
            stack.push(s[i])
        }
    }
    let newStack = []
    for (let i = 0; i < stack.length; i++) {
        if (newStack[newStack.length - 1] === secondCompose[0] && stack[i] === secondCompose[1]) {
            newStack.pop()
            score += lowerScore
        } else {
            newStack.push(stack[i])
        }
    }
    return score
};
```
空間複雜度：O(n)
時間複雜度：O(n)
