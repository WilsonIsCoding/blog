## 題解
```javascript
/**
 * @param {number[]} coins
 * @param {number} amount
 * @return {number}
 */
var coinChange = function (coins, amount) {
  let maxAmount = amount + 1;
  let dp = new Array(maxAmount).fill(maxAmount);
  dp[0] = 0;
  for (const coin of coins) {
    for (let i = coin; i < maxAmount; i++) {
      dp[i] = Math.min(dp[i], dp[i - coin] + 1);
    }
  }
  return dp[amount] === maxAmount ? -1 : dp[amount];
};
```
時間複雜度：O(coins.length * amount.length)
空間複雜度：O(Array.length)
