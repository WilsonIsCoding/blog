## 題解

```javascript
/**
 * @param {string[]} folder
 * @return {string[]}
 */
var removeSubfolders = function (folder) {
    folder = folder.sort(); //O(nlogn)
    let res = [];
    for (const fold of folder) {//O(n)
        if (!res[res.length - 1] || !fold.startsWith(res[res.length - 1] + '/')) {
            res.push(fold)
        }
    }
    return res
};
```
空間複雜度：O(nlogn+n)=>O(n)
時間複雜度：O(nlogn)
