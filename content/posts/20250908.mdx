## 題解

```javascript
/**
 * @param {number} n
 * @return {number[]}
 */
var getNoZeroIntegers = function (n) {
    for (let i = 1; i < n; i++) {
        let remain = n - i;
        if (remain.toString().indexOf(0) == -1 && i.toString().indexOf(0) == -1) {
            return [remain, i]
        }
    }
    return []
};
```

時間複雜度：O(n)
空間複雜度：O(1)
