## 題解

```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */
var longestSubarray = function (nums) {
    let right = 0;
    let left = 0;
    let zero = 0;
    let result = 0
    while (right < nums.length) {
        if (nums[right] === 0) {
            zero++
        }
        while (zero > 1) {
            if (nums[left] === 0) {
                zero--
            }
            left++
        }
        result = Math.max(result, right - left)
        right ++
    }
    return result;
};
```

時間複雜度：O(n)
空間複雜度：O(4)
