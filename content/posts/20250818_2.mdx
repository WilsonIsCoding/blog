## 題解

```javascript
/**
 * @param {number[]} nums
 * @return {number[]}
 */
var sortArrayByParityII = function (nums) {
    let result = new Array(nums.length);
    let odd = 0;
    let even = 1;
    for (let i = 0; i < nums.length; i++) {
        if (nums[i] % 2 === 0) {
            result[odd] = nums[i]
            odd += 2
        } else {
            result[even] = nums[i]
            even += 2
        }
    }
    return result
};
```

時間複雜度：O(n)
空間複雜度：O(n)
