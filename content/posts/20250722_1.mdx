## 題解

```javascript
/**
 * @param {string} s
 * @return {number}
 */
var lengthOfLongestSubstring = function (s) {
    let count = 0;
    let left = 0;
    let right = 0;
    let total = 0;
    let set = new Set();
    while (right < s.length) {
        while (set.has(s[right])) {
            set.delete(s[left])
            count--
            left++
        }
        set.add(s[right])
        right++
        count++
        total = Math.max(total, count)
    }
    return total
};
```
空間複雜度：O(n)
時間複雜度：O(n)
