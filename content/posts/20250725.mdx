## 題解

```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxSum = function (nums) {
    nums = nums.sort((a, b) => a - b);
    if (nums[nums.length - 1] < 0) return nums[nums.length - 1]
    let set = new Set(nums);
    let count = 0
    set.forEach((value) => {
        if (value > 0) {
            count += value
        }
    })
    return count
};
```
空間複雜度：O(n)
時間複雜度：O(n)
