## 題解

```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */
var maximumUniqueSubarray = function (nums) {
    let left = 0;
    let right = 0;
    let currentTotal = 0;
    let total = 0;
    let set = new Set()
    while (right < nums.length) {
        while (set.has(nums[right])) {
            currentTotal -= nums[left]
            set.delete(nums[left])
            left++
        }

        set.add(nums[right])
        currentTotal += nums[right]
        total = Math.max(currentTotal, total)
        right++
    }
    return total
};
```
空間複雜度：O(n)
時間複雜度：O(n)
