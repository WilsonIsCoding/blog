## 題解

```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */
var countHillValley = function (nums) {
    let direction = null;
    let count = 0;
    let currentNum = nums[0];
    for (let i = 1; i < nums.length; i++) {
        if (currentNum < nums[i] && direction === 2) {
            direction = 1
            count++
        } else if (currentNum < nums[i]) {
            direction = 1
        } else if (currentNum > nums[i] && direction === 1) {
            direction = 2
            count++
        } else if (currentNum > nums[i]) {
            direction = 2
        }
        currentNum = nums[i]
    }
    return count
};
```
空間複雜度：O(n)
時間複雜度：O(n)
