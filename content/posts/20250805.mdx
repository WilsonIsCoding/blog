## 題解
```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */
var longestNiceSubarray = function (nums) {
    let mask = 0;
    let maxCount = 0;
    let left = 0;
    let right = 0;
    while (right < nums.length) {
        if ((mask & nums[right]) === 0) {
            mask |= nums[right]
            right++
        } else {
            while ((mask & nums[right]) != 0) {
                mask ^= nums[left]
                left++
            }
        }
        maxCount = Math.max(maxCount, right - left)
    }
    return maxCount
};
```
空間複雜度：O(1)
時間複雜度：O(n)
